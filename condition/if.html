    # if

    Nimのifはif-elif-elseのような構文で書きます。

    var
  a = "apple"
  b = "banana"

if a == b:
  echo "apple is banana :)"
elif a != b:
  echo "appppple 🍎"
  echo "banaaana 🍌"
else:
  echo "See you"
